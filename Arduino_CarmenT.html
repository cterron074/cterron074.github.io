Estación de
Monitoreo Ambiental

Proyecto de Arduino Realizado con un compañero de clase, el cual consiste en un Sistema embebido que mide, registra y visualiza en tiempo real la humedad del suelo, la luz ambiental y el CO₂ del entorno mediante tres sensores conectados a un Arduino. Todas estas medidas se almacenan y se reflejan por pantallas LCD 2x16.

Todo este proyecto fue expuesto en el Stand del instituto I.E.S. Lope de Vega en la feria de 'Madrid Es Ciencia 2026'. Formó parte de la colaboración entre los alumnos de FP superior de Informática y los de FP superior de Ciencias. En este proyecto aprendimos a integrar hardware y software, a calibrar sensores, a gestionar datos en tiempo real y a presentar resultados de forma clara y atractiva. Fue una experiencia muy enriquecedora que nos permitió aplicar nuestros conocimientos en un contexto práctico y real.

Tres sensores, un sistema

Cada sensor captura una dimensión clave del entorno. Los datos se muestrean cada segundo y se escriben en la tarjeta SD con marca de tiempo.

🌱

Humedad de Suelo

// Sensor: YL-69 + YL-38

Mide la resistencia eléctrica del sustrato. A mayor humedad, menor resistencia. La señal analógica se normaliza a un porcentaje de 0–100 %.

SeñalAnalógica
PinA0
Rango0 – 100 %
Tensión3.3 – 5 V

Luz Ambiental

// Sensor: BH1750 (I²C)

Fotorresistor digital con resolución de 1 lux. Comunica vía I²C y entrega luxes directamente, sin necesidad de calibración manual.

SeñalI²C
Dirección0x23
Rango0 – 65535 lx
Resolución1 lux

CO₂ Ambiental

// Sensor: MQ-135

Sensor electroquímico sensible al dióxido de carbono y otros gases. Salida analógica convertida a ppm mediante ecuación de calibración.

Espera 1 s

// Adquisición de datos

  • Lectura analógica del YL-69 (pin A0)
  • Solicitud I²C al BH1750 (0x23)
  • Lectura analógica del MQ-135 (pin A1)
  • Marca de tiempo con módulo RTC DS1307

// Procesamiento

  • Mapeo ADC 0–1023 → 0–100 % (suelo)
  • Conversión I²C raw → lux (BH1750)
  • Cálculo ppm CO₂ con curva de calibración
  • Promedio móvil de 5 muestras

// Visualización

  • Pantalla OLED SSD1306 128×64 (I²C)
  • Rotación de las 3 lecturas cada 3 s
  • LED de alerta si CO₂ > 1000 ppm
  • Serial monitor a 9600 baud

// Almacenamiento

  • Módulo SD SPI (pines 10–13)
  • Archivo data.csv con cabecera
  • Columnas: timestamp, soil_pct, lux, co2_ppm
  • Cierre seguro del archivo tras cada escritura

ECOSO - Arduino Station

Despliega el bloque para examinar o modificar el sketch de Arduino.

ARDUINO C++ 
        #include <Wire.h>
        #include <Adafruit_SGP30.h>
        #include <LiquidCrystal_I2C.h>

        // Configuración de objetos
        Adafruit_SGP30 sgp;
        LiquidCrystal_I2C lcd(0x27, 16, 2);

        // Pines y variables
        const int pinSensorLuz = A1;
        const int pinHumedadSuelo = A2;

        void setup() {
          Serial.begin(115200);
          Wire.begin();
        
          lcd.init();
          lcd.backlight();
          lcd.print("Iniciando...");
        
          delay(1000);
          lcd.clear();

          Serial.begin(115200);
          delay(2000);
          Serial.println("SGP30 test");
          Wire.begin();
          Wire.setClock(100000);

          // Inicialización I2C
          Wire.begin();
        
          // Inicialización SGP30
          if (!sgp.begin()){
            Serial.println("ERROR: No se pudo encontrar el sensor SGP30.");
            Serial.println("Verifica que esté conectado a 5V (NO a Vin) y los pines SDA/SCL.");
            while (1); // Detener el programa si falla
          }

          Serial.println("Sensor SGP30 detectado correctamente.");
          Serial.println("Leyendo datos cada segundo...\n");      

          Serial.print("Número de serie: ");
          Serial.print(sgp.serialnumber[0], HEX);
          Serial.print(sgp.serialnumber[1], HEX);
          Serial.println(sgp.serialnumber[2], HEX);
        }

        void loop() {
          lcd.print("¡WELCOME,HI!");
          Serial.println("--- ESTACIÓN DE MONITOREO INICIADA ---");
          if (! sgp.begin()){
            Serial.println("Sensor SGP30 no encontrado");
            while (1);
          }
          if (! sgp.IAQmeasure()) {
            Serial.println("Error en medición");
            return;
          }
          // 2. Lectura de Luz (A1)
          int luzRaw = analogRead(pinSensorLuz);
          int porcLuz = map(luzRaw, 1023, 0, 0, 100);
          porcLuz = constrain(porcLuz, 0, 100);
          // 3. Lectura de Humedad de Suelo (A2)
          delay(3000);
          int sueloRaw = analogRead(pinHumedadSuelo);
          int porcSuelo = map(sueloRaw, 1023, 250, 0, 100);
          porcSuelo = constrain(porcSuelo, 0, 100);
          delay(3000);
          // Luz
          Serial.print("LUZ   -> "); Serial.print(porcLuz); Serial.print("% ");
          if (porcLuz < 30) {
            Serial.println("(Ambiente Oscuro)");
            lcd.println("Oscuro");
          }
          else {
            Serial.println("(Ambiente Claro)");
            lcd.println("Claro");
          }
          // Suelo
          Serial.print("SUELO -> "); Serial.print(porcSuelo); Serial.print("%");
          if (porcSuelo < 20){ Serial.println("(NECESITA AGUA)");
            lcd.println("+Agua xfi");
          }
          else{
            Serial.println("(Humedad OK)");
            lcd.println("Humedad ok");
          }

          delay(3000);
          Serial.print("eCO2: ");
          Serial.print(sgp.eCO2);
          Serial.print(" ppm\t");

          lcd.print();
          Serial.print("TVOC: ");
          Serial.print(sgp.TVOC);
          Serial.println(" ppb");

          delay(2000);
        }

// ────────────────────────────────────────────────── #include <Wire.h> #include <SPI.h> #include <SD.h> #include <BH1750.h> #include <Adafruit_SSD1306.h> #include <RTClib.h> // ── Pines ─────────────────────────────────────────── const int PIN_SUELO = A0; const int PIN_CO2 = A1; const int PIN_LED = 13; const int PIN_SD_CS = 10; // ── Objetos ───────────────────────────────────────── BH1750 lightMeter; Adafruit_SSD1306 oled(128, 64, &Wire, -1); RTC_DS1307 rtc; File logFile; // ── Variables globales ─────────────────────────────── float soilPct = 0; float luxVal = 0; float co2Ppm = 0; uint8_t oledPage = 0; unsigned long lastOledSwitch = 0; void setup() { Serial.begin(9600); Wire.begin(); pinMode(PIN_LED, OUTPUT); lightMeter.begin(); oled.begin(SSD1306_SWITCHCAPVCC, 0x3C); oled.clearDisplay(); oled.setTextColor(WHITE); rtc.begin(); if (!rtc.isrunning()) rtc.adjust(DateTime(__DATE__, __TIME__)); if (!SD.begin(PIN_SD_CS)) { Serial.println("ERROR: SD no detectada"); while (1); } logFile = SD.open("data.csv", FILE_WRITE); if (logFile) { logFile.println("timestamp,soil_pct,lux,co2_ppm"); logFile.close(); } Serial.println("EcoStation lista ✓"); } void readSensors() { int raw = 0; for (int i = 0; i < 5; i++) raw += analogRead(PIN_SUELO); soilPct = map(raw / 5, 1023, 0, 0, 100); luxVal = lightMeter.readLightLevel(); float v = analogRead(PIN_CO2) * (5.0 / 1023.0); float rs = ((5.0 - v) / v) * 10.0; co2Ppm = 116.6020682 * pow(rs / 76.63, -2.777); } void saveToSD() { DateTime now = rtc.now(); logFile = SD.open("data.csv", FILE_WRITE); if (logFile) { logFile.print(now.timestamp(DateTime::TIMESTAMP_FULL)); logFile.print(","); logFile.print(soilPct, 1); logFile.print(","); logFile.print(luxVal, 1); logFile.print(","); logFile.println(co2Ppm, 1); logFile.close(); } } void updateOLED() { if (millis() - lastOledSwitch > 3000) { oledPage = (oledPage + 1) % 3; lastOledSwitch = millis(); } oled.clearDisplay(); oled.setTextSize(1); oled.setCursor(0, 0); if (oledPage == 0) { oled.print("SUELO: "); oled.print(soilPct); oled.println("%"); } else if (oledPage == 1) { oled.print("LUZ: "); oled.print(luxVal); oled.println(" lx"); } else { oled.print("CO2: "); oled.print(co2Ppm); oled.println(" ppm"); } oled.display(); } void loop() { readSensors(); updateOLED(); saveToSD(); digitalWrite(PIN_LED, co2Ppm > 1000 ? HIGH : LOW); Serial.print("Suelo: "); Serial.print(soilPct); Serial.print("% | Lux: "); Serial.print(luxVal); Serial.print(" | CO2: "); Serial.print(co2Ppm); Serial.println(" ppm"); delay(1000); }
